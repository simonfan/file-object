'use strict';

var subject = require('subject'),
	_ = require('lodash');

var file = require('../index');

/**
 * Constructor for multiple file reader.
 *
 * @class files
 * @param fpaths {Array|Object|-String-}
 *     List of paths to the files to be read in this object.
 *     {Object} -> keyed by file id (any String value), valued by file path
 *     {Array} -> multiple file paths
 *     -TO_IMPLEMENT- {String} -> minimatch pattern
 */
var files = module.exports = subject(function files(fpaths, options) {
	this.fpaths = fpaths;

	/**
	 * Build the file objects and keep reference to them in an object.
	 */
	this.files = {};

	if (_.isArray(fpaths)) {

		fpaths.forEach(function (fpath) {
			this.files[fpath] = this.file(fpath);
		}.bind(this));

	} else {

		_.each(fpaths, function (fpath, fid) {
			// build the file object using a prototype property,
			// so that it is extendable.
			this.files[fid] = this.file(fpath);
		}.bind(this));

	}
});

files.proto({
	data: function (fid, fdata) {

		if (arguments.length === 1) {

			_.each(fid, function (fdata, fid) {
				this.data(fid, fdata);
			}.bind(this));

		} else if (arguments.length === 2) {

			// set the data of a single file.
			this.file(fid).data(fdata);
		}
	}
});

/**
 * The single file builder. Overwrite this for custom file object behaviour.
 * @property/method
 */
files.proto('file', file);

files.proto(require('./read'));

files.proto(require('./write'));

files.proto(require('./unlink'));
